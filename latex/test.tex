\begin{subsection}{程式語言}
\begin{frame}{程式語言}
\begin{itemize}
\item 機器語言(Machine Language)
  \begin{itemize}
  \item 電腦使用０與１所組成的語言。
  \end{itemize}
  \item 高階程式語言
  \begin{itemize}
  \item 人比較可以看的懂的語言，用英文與數學符號所構成。
  \item 很好轉換成機器語言，藉此成為人與電腦溝通的橋梁。
  \end{itemize}
  \end{itemize}

  \begin{frame}{編譯器(Compiler)}
  \begin{itemize}
